codigo html para formulario:

 <form id="form" action="" class="form">
            <h3 class="tag"><span>Contactenos</span></h3>
            <label for="name">Nombre y apellido<span>*</span></label>
            <input name="name" required type="text" id="name">
            <label for="email">Correo electronico<span>*</span></label>
            <input name="email" type="text" id="email" required>
            <label for="message">Mensaje</label>
            <textarea id="message" name="message" cols="30" rows="10"></textarea>
            <button type="submit" class="btn-primary-full">Enviar</button>
        </form>
        <a href="mailto:elifredmason@gmail.com" id="opacidad">mail</a>

         <script>
        const $form = document.querySelector('#form')
        const $buttonMailto = document.querySelector('#opacidad')

        $form.addEventListener('submit', handleSubmit)

        function handleSubmit(event) {
            event.preventDefault()
            const form = new FormData(this)
           // console.log(form.get('name')) 
            $buttonMailto.setAttribute('href', `mailto:elifredmason@gmail.com?subject=${form.get('name')}${form.get('email')}&body=${form.get('message')}`)
            $buttonMailto.click()
        }
    </script>
